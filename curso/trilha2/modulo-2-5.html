🐚 ShellTool sandboxed

import asyncio
ALLOWED = {"ls", "grep", "cat", "find", "head", "tail", "wc"}

async def execute(self, command: str):
    binary = command.split()[0]
    if binary not in ALLOWED:
        return ToolResult(error=f"comando '{binary}' não autorizado", output=None)
    try:
        proc = await asyncio.create_subprocess_shell(
            command, stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE,
            cwd="/tmp/workspace")
        stdout, stderr = await asyncio.wait_for(proc.communicate(), timeout=30)
        return ToolResult(output=stdout.decode()[:5000], error=stderr.decode() if proc.returncode else None)
    except asyncio.TimeoutError:
        proc.kill()
        return ToolResult(error="timeout 30s", output=None)

🌐 WebFetchTool

import httpx, html2text

class WebFetchTool(Tool):
    name = "web_fetch"
    description = "Baixa URL e retorna conteúdo em markdown limpo. Use pra ler artigos, docs, páginas públicas."
    parameters = {"type": "object", "properties": {"url": {"type": "string"}}, "required": ["url"]}

    async def execute(self, url):
        async with httpx.AsyncClient(timeout=15, follow_redirects=True,
            headers={"User-Agent": "Mozilla/5.0"}) as cli:
            r = await cli.get(url)
            md = html2text.html2text(r.text)
            return ToolResult(output=md[:10000], error=None)

📅 CalendarTool

from google.oauth2.credentials import Credentials
from googleapiclient.discovery import build

class CalendarTool(Tool):
    name = "calendar_list"
    description = "Lista eventos do dia. Use quando user pergunta sobre agenda."
    parameters = {"type": "object", "properties": {
        "date": {"type": "string", "description": "YYYY-MM-DD. Use today se omitido."}
    }}
    async def execute(self, date=None):
        creds = Credentials.from_authorized_user_file('token.json')
        svc = build('calendar', 'v3', credentials=creds)
        events = svc.events().list(calendarId='primary', timeMin=...).execute()
        return ToolResult(output="\\n".join(f"- {e['summary']} @ {e['start']}" for e in events.get('items', [])))

📝 FileTool chroot

from pathlib import Path

class FileWriteTool(Tool):
    def __init__(self, workspace):
        self.workspace = Path(workspace).resolve()

    async def execute(self, path, content):
        target = (self.workspace / path).resolve()
        if not target.is_relative_to(self.workspace):
            return ToolResult(error="path fora do workspace", output=None)
        if len(content) > 100_000:
            return ToolResult(error="conteúdo > 100k chars", output=None)
        target.parent.mkdir(parents=True, exist_ok=True)
        target.write_text(content)
        return ToolResult(output=f"escrito {len(content)} bytes em {path}")

📜 Padrões pra LLM entender

💡 4 regras

  1. Description começa com verbo: "Roda...", "Busca...", "Cria...".
  2. Mostre 1 exemplo quando faz sentido: "Ex: ls -la /tmp".
  3. Diga quando NÃO usar: "Não use pra arquivos > 1MB".
  4. Parameters < 7. Se passa, simplifica ou divide tool.

📝 Resumo

4 tools cobrem 80% dos casos: shell, web, calendar, file. Descrição boa = 90% accuracy do LLM.
