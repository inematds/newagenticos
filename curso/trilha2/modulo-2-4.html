📊 SQLite FTS5 — implementação

import aiosqlite, uuid, time
from .base import MemoryStore, MemoryEntry

class SQLiteFTS5Store(MemoryStore):
    def __init__(self, path):
        self.path = path

    async def setup(self):
        async with aiosqlite.connect(self.path) as db:
            await db.execute("""
                CREATE VIRTUAL TABLE IF NOT EXISTS memories USING fts5(
                    id, content, category, user_id, created_at UNINDEXED
                )""")

    async def save(self, content, category, user_id):
        eid = str(uuid.uuid4())
        async with aiosqlite.connect(self.path) as db:
            await db.execute("INSERT INTO memories VALUES (?,?,?,?,?)",
                (eid, content, category, user_id, time.time()))
            await db.commit()
        return eid

    async def search(self, query, user_id, limit=10):
        async with aiosqlite.connect(self.path) as db:
            cursor = await db.execute(
                "SELECT id, content, category, user_id, created_at FROM memories "
                "WHERE memories MATCH ? AND user_id = ? ORDER BY rank LIMIT ?",
                (query, user_id, limit))
            rows = await cursor.fetchall()
            return [MemoryEntry(*r) for r in rows]

🔮 Vector store (quando FTS falha)

FTS5 não acha "automóvel" buscando "carro". Vector store sim. Mas só vale a partir de ~10k entries OU quando vocabulário do user diverge muito.

# Chroma local (gratuito)
import chromadb

collection = client.get_or_create_collection("memories")
collection.add(documents=[content], metadatas=[{"user_id": uid}], ids=[eid])
results = collection.query(query_texts=[query], where={"user_id": uid}, n_results=10)

🗑️ Forget e LGPD

async def forget(self, entry_id):
    async with aiosqlite.connect(self.path) as db:
        # Audit antes (sem o conteúdo)
        await db.execute("INSERT INTO audit (op, entry_id, ts) VALUES (?,?,?)",
            ("forget", entry_id, time.time()))
        await db.execute("DELETE FROM memories WHERE id = ?", (entry_id,))
        await db.commit()

📝 Resumo

SQLite FTS5 é o "default certo" pra 90% dos casos. Vector só quando FTS falhar. Forget é obrigatório.
