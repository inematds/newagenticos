🔁 Long-polling explicado

📨 TelegramChannel completo

import httpx, asyncio, time
from .base import Channel, IncomingMessage, OutgoingMessage

class TelegramChannel(Channel):
    def __init__(self, token, allowed_user_ids=None):
        self.token = token
        self.allowed = set(allowed_user_ids or [])
        self.base = f"https://api.telegram.org/bot{token}"
        self.offset = 0
        self._stop = False

    async def start(self, on_message):
        async with httpx.AsyncClient(timeout=35) as cli:
            while not self._stop:
                r = await cli.get(f"{self.base}/getUpdates",
                    params={"offset": self.offset, "timeout": 30})
                for up in r.json().get("result", []):
                    self.offset = up["update_id"] + 1
                    msg = up.get("message", {})
                    uid = msg.get("from", {}).get("id")
                    if self.allowed and uid not in self.allowed:
                        continue
                    await on_message(IncomingMessage(
                        user_id=str(uid),
                        text=msg.get("text", ""),
                        timestamp=time.time(),
                        metadata={"chat_id": msg["chat"]["id"]}))

    async def send(self, msg):
        async with httpx.AsyncClient(timeout=10) as cli:
            await cli.post(f"{self.base}/sendMessage",
                json={"chat_id": int(msg.user_id), "text": msg.text,
                      "parse_mode": "MarkdownV2"})

    async def stop(self):
        self._stop = True

📝 MarkdownV2 escaping

Telegram exige escape de caracteres reservados (_*[]()~`>#+-=|{}.!).

SPECIALS = r'_*[]()~`>#+-=|{}.!'
def escape_md(text):
    return ''.join('\\\\' + c if c in SPECIALS else c for c in text)
