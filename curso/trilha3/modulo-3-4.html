🛑 Rate limit + circuit breaker

from collections import defaultdict
import time

class TokenBucket:
    def __init__(self):
        self.buckets = defaultdict(lambda: {"tokens": 0, "last": time.time()})

    def allow(self, user_id, capacity, refill_per_sec):
        b = self.buckets[user_id]
        now = time.time()
        b["tokens"] = min(capacity, b["tokens"] + (now - b["last"]) * refill_per_sec)
        b["last"] = now
        if b["tokens"] >= 1:
            b["tokens"] -= 1
            return True
        return False
