Pasta dedicada por user UUID. memory.db, files/, preferences.json isolados.

from pathlib import Path

def user_workspace(user_id: str) -> Path:
    base = Path.home() / ".app" / "users" / user_id
    base.mkdir(parents=True, exist_ok=True)
    return base

# Memory store por user
memory = SQLiteFTS5Store(user_workspace(user.id) / "memory.db")
